Widget Themes

Dynamic Context API

// Update context when user navigates
BabbleBuddy.setContext({
  app: 'My App',
  page: 'checkout',
  role: 'support',
  instructions: 'Help with payment and shipping questions.',
  schema: ['cart', 'shipping', 'payments'],
  user: { name: 'John', plan: 'premium' }
});

Embed in Your App

<script src="YOUR_WIDGET_URL/babble-buddy.umd.cjs"></script>
<script>
  BabbleBuddy.init({
    appToken: 'your-app-token',
    apiUrl: 'your-agent-core-url',
    context: {
      app: 'My App',
      page: 'home',
      instructions: 'Help users with...'
    }
  });
</script>
